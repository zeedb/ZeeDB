select exists (select store_id from store where store.store_id = customer.store_id) as ok 
from customer

LogicalMap $query.ok:$exists
	LogicalJoin Mark $exists (Is customer.store_id store_id)
		LogicalDependentJoin [store_id]
			LogicalMap store.store_id
				LogicalFilter (Equal store.store_id store_id)
					LogicalGet store
			LogicalGet customer
		LogicalGet customer
