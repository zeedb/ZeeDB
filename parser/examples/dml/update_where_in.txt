set account_number = 0 
where person_id in (select person_id from person where first_name = "Joe")

LogicalUpdate customer
	LogicalMap customer.customer_id customer.person_id customer.store_id customer.account_number:0 customer.modified_date $tid
		LogicalFilter $in
			LogicalJoin Mark $in (Equal customer.person_id person.person_id)
				LogicalMap person.person_id
					LogicalFilter (Equal person.first_name "Joe")
						LogicalGet person
				LogicalGet customer
