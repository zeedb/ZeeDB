from person 
where person_id not in (select person_id from customer)

	LogicalFilter (Not $mark.$in)
		LogicalJoin Mark $mark.$in (Equal person.person_id customer.person_id)
			LogicalMap customer.person_id
				Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)

			Filter* (LessOrEqual person.$xmin (Xid)) (Less (Xid) person.$xmax)
