from customer

	LogicalJoin Mark $mark.$in (Equal customer.store_id $domain.store_id) (Equal 1 store.store_id)
		LogicalDependentJoin $domain.store_id

				LogicalFilter (Equal store.store_id $domain.store_id)
					Map* store.store_id
						LogicalGet store
			LogicalProject customer.store_id:$domain.store_id
				Map* customer.store_id
					LogicalGet customer
		Map* customer.store_id
