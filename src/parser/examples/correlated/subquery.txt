from customer

LogicalMap $query.$col1:store.name
	LogicalDependentJoin Single customer.store_id
		LogicalMap store.name
			LogicalFilter (Equal store.store_id customer.store_id)
				Map* store.store_id store.name
					LogicalGet store
		Map* customer.store_id
			LogicalGet customer
