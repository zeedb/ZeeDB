select (select name from store where store.store_id = customer.store_id) 
from customer

LogicalMap $query.$col1:store.name
	LogicalJoin Single (Equal customer.store_id customer.store_id')
		LogicalDependentJoin [customer.store_id']
			LogicalMap store.name
				LogicalFilter (Equal store.store_id customer.store_id')
					Filter* (LessOrEqual store.$xmin (Xid)) (Less (Xid) store.$xmax)
						Map* store.store_id store.name store.$xmin store.$xmax
							LogicalGet store
			Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
				Map* customer.store_id' customer.$xmin customer.$xmax
					LogicalGet customer
		Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
			Map* customer.store_id customer.$xmin customer.$xmax
				LogicalGet customer
