select exists (select store_id from store where store.store_id = customer.store_id) as ok 
from customer

LogicalMap $query.ok:$mark.$exists
	LogicalJoin Mark $mark.$exists (Equal customer.store_id customer.store_id')
		LogicalDependentJoin [customer.store_id']
			LogicalMap store.store_id
				LogicalFilter (Equal store.store_id customer.store_id')
					Filter* (LessOrEqual store.$xmin (Xid)) (Less (Xid) store.$xmax)
						Map* store.store_id store.$xmin store.$xmax
							LogicalGet store
			Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
				Map* customer.store_id' customer.$xmin customer.$xmax
					LogicalGet customer
		Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
			Map* customer.store_id customer.$xmin customer.$xmax
				LogicalGet customer
