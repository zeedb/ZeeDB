select customer_id 
from customer 
order by modified_date 
limit 100 
offset 10

LogicalLimit 100 10
	LogicalSort customer.modified_date
		Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
			Map* customer.customer_id customer.modified_date customer.$xmin customer.$xmax
				LogicalGet customer
