select store_id 
from customer 
group by 1 
having count(*) > 1

LogicalMap $groupby.store_id
	LogicalFilter (Greater $aggregate.$agg1 1)
		LogicalAggregate $groupby.store_id $aggregate.$agg1:(CountStar)
			LogicalMap $groupby.store_id:customer.store_id
				Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
					Map* customer.store_id customer.$xmin customer.$xmax
						LogicalGet customer
