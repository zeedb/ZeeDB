select customer.customer_id, store.store_id 
from customer, store

LogicalMap customer.customer_id store.store_id
	LogicalJoin Inner
		Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
			Map* customer.customer_id customer.$xmin customer.$xmax
				LogicalGet customer
		Filter* (LessOrEqual store.$xmin (Xid)) (Less (Xid) store.$xmax)
			Map* store.store_id store.$xmin store.$xmax
				LogicalGet store
