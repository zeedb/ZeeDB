select 1 
from person, customer 
where person.person_id = customer.person_id and customer.store_id in (select store_id from store)

LogicalMap $query.$col1:1
	LogicalFilter (Equal person.person_id customer.person_id) $mark.$in
		LogicalJoin Mark $mark.$in (Equal customer.store_id store.store_id)
			LogicalMap store.store_id
				Filter* (LessOrEqual store.$xmin (Xid)) (Less (Xid) store.$xmax)
					Map* store.store_id store.$xmin store.$xmax
						LogicalGet store
			LogicalJoin Inner
				Filter* (LessOrEqual person.$xmin (Xid)) (Less (Xid) person.$xmax)
					Map* person.person_id person.$xmin person.$xmax
						LogicalGet person
				Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
					Map* customer.person_id customer.store_id customer.$xmin customer.$xmax
						LogicalGet customer
