with foo as (select customer_id from customer) select customer_id 
from foo

LogicalWith foo
	LogicalMap customer.customer_id
		Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
			Map* customer.customer_id customer.$xmin customer.$xmax
				LogicalGet customer
	LogicalMap foo.customer_id
		LogicalGetWith foo
