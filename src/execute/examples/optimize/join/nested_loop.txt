select customer.customer_id, store.store_id
from customer, store

Map customer.customer_id store.store_id
	NestedLoop Inner
		Filter* (LessOrEqual $xmin (Xid)) (Less (Xid) $xmax)
			Map* store.store_id $xmin $xmax
				SeqScan store
		Filter* (LessOrEqual $xmin (Xid)) (Less (Xid) $xmax)
			Map* customer.customer_id $xmin $xmax
				SeqScan customer
