select customer.customer_id, store.store_id
from customer
full outer join store on customer.store_id = store.store_id

Map customer.customer_id store.store_id
	HashJoin Outer (Equal customer.store_id store.store_id)
		Partition* customer.store_id
			Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
				Map* customer.customer_id customer.store_id customer.$xmin customer.$xmax
					SeqScan customer
		Partition* store.store_id
			Filter* (LessOrEqual store.$xmin (Xid)) (Less (Xid) store.$xmax)
				Map* store.store_id store.$xmin store.$xmax
					SeqScan store
