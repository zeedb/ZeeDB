update customer
set account_number = account_number + 1
from person
where customer.person_id = person.person_id

Update customer.account_number:$update.$expr
	Map $update.$expr:(Add customer.account_number 1)
		HashJoin Inner (Equal customer.person_id person.person_id)
			Partition* customer.person_id
				Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
					Map* customer.person_id customer.account_number customer.$xmin customer.$xmax
						SeqScan customer
			Partition* person.person_id
				Filter* (LessOrEqual person.$xmin (Xid)) (Less (Xid) person.$xmax)
					Map* person.person_id person.$xmin person.$xmax
						SeqScan person
