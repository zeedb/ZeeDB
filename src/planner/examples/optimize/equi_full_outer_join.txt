select customer.customer_id, store.store_id 
from customer 
full outer join store on customer.store_id = store.store_id

LogicalProject customer.customer_id store.store_id
	LogicalJoin Outer (Equal customer.store_id store.store_id)
		Project* customer.customer_id customer.store_id
			LogicalGet customer
		Project* store.store_id
			LogicalGet store

Project customer.customer_id store.store_id
	HashJoin Outer customer.store_id=store.store_id
		Project* customer.customer_id customer.store_id
			SeqScan customer
		Project* store.store_id
			SeqScan store
