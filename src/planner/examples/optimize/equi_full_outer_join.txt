select customer.customer_id, store.store_id 
from customer 
full outer join store on customer.store_id = store.store_id

Map customer.customer_id store.store_id
	HashJoin Outer customer.store_id=store.store_id
		Map* customer.customer_id customer.store_id
			SeqScan customer
		Map* store.store_id
			SeqScan store
