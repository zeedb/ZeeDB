from person, customer
where person.person_id = customer.person_id
and customer.store_id in (select store_id from store)

	HashJoin Semi store.store_id=customer.store_id
		NestedLoop Inner

				NestedLoop Inner

					Map* person.person_id
						SeqScan person
			Map* store.store_id
				SeqScan store
		HashJoin Inner customer.person_id=person.person_id
			Map* customer.person_id customer.store_id
				SeqScan customer
			Map* person.person_id
				SeqScan person
