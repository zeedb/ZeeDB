select first_name 
from person 
where person_id in (select person_id from customer)

LogicalProject person.first_name
	LogicalJoin Semi (Equal person.person_id customer.person_id)
		Project* customer.person_id
			LogicalGet customer
		Project* person.person_id person.first_name
			LogicalGet person

Project person.first_name
	HashJoin Semi customer.person_id=person.person_id
		Project* customer.person_id
			SeqScan customer
		Project* person.person_id person.first_name
			SeqScan person
