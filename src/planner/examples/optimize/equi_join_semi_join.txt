select 1 
from person, customer 
where person.person_id = customer.person_id 
and customer.store_id in (select store_id from store)

LogicalProject $query.$col1:1
	LogicalFilter (Equal person.person_id customer.person_id)
		LogicalJoin Semi (Equal customer.store_id store.store_id)
			LogicalProject store.store_id
				LogicalGet store
			LogicalJoin Inner
				LogicalGet person
				LogicalGet customer
