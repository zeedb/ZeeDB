select customer.customer_id, store.store_id 
from customer, store 
where customer.store_id = store.store_id

LogicalMap customer.customer_id store.store_id
	LogicalJoin Inner (Equal customer.store_id store.store_id)
		Map* customer.customer_id customer.store_id
			LogicalGet customer
		Map* store.store_id
			LogicalGet store

Map customer.customer_id store.store_id
	HashJoin Inner store.store_id=customer.store_id
		Map* store.store_id
			SeqScan store
		Map* customer.customer_id customer.store_id
			SeqScan customer
