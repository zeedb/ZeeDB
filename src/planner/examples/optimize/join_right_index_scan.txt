select customer.customer_id, store.store_id 
from customer, store 
where store.store_id = 1

LogicalMap customer.customer_id store.store_id
	LogicalJoin Inner
		Map* customer.customer_id
			LogicalGet customer
		Map* store.store_id
			Filter* (Equal store.store_id 1)
				LogicalGet store

Map customer.customer_id store.store_id
	NestedLoop Inner
		Map* store.store_id
			IndexScan store(1:store.store_id)
		Map* customer.customer_id
			SeqScan customer
