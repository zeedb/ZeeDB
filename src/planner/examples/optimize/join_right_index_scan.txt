select customer.customer_id, store.store_id 
from customer, store 
where store.store_id = 1

LogicalProject customer.customer_id store.store_id
	LogicalJoin Inner
		Project* customer.customer_id
			LogicalGet customer
		Project* store.store_id
			Filter* (Equal store.store_id 1)
				LogicalGet store

Project customer.customer_id store.store_id
	NestedLoop Inner
		Project* store.store_id
			IndexScan store(1:store.store_id)
		Project* customer.customer_id
			SeqScan customer
