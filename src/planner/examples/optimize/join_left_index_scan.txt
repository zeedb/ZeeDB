select customer.customer_id, store.store_id 
from customer, store 
where customer.customer_id = 1

LogicalMap customer.customer_id store.store_id
	LogicalJoin Inner
		Map* customer.customer_id
			Filter* (Equal customer.customer_id 1)
				LogicalGet customer
		Map* store.store_id
			LogicalGet store

Map customer.customer_id store.store_id
	NestedLoop Inner
		Map* customer.customer_id
			IndexScan customer(1:customer.customer_id)
		Map* store.store_id
			SeqScan store
