select customer.customer_id, store.store_id
from customer, store
where customer.store_id = store.store_id
and customer.customer_id = 1

Map customer.customer_id store.store_id
	NestedLoop Inner (Equal customer.store_id store.store_id)
		Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax) (Equal customer.customer_id 1)
			Map* customer.customer_id customer.store_id customer.$xmin customer.$xmax
				SeqScan customer
		Filter* (LessOrEqual store.$xmin (Xid)) (Less (Xid) store.$xmax)
			Map* store.store_id store.$xmin store.$xmax
				SeqScan store
