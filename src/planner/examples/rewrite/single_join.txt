select (select name from store where store.store_id = customer.customer_id and store.name like "A%"), (select first_name from person where person.person_id = customer.person_id) from customer

LogicalProject $query.$col1:store.name $query.$col2:person.first_name
	LogicalSingleJoin (Equal person.person_id customer.person_id)
		LogicalGet person
		LogicalSingleJoin (Equal store.store_id customer.customer_id)
			LogicalFilter (Like store.name A%)
				LogicalGet store
			LogicalGet customer
