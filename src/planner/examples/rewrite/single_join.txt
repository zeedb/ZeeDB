select (select name from store where store.store_id = customer.customer_id and store.name like "A%"), (select first_name from person where person.person_id = customer.person_id) from customer

LogicalProject $query.$col1:store.name $query.$col2:person.first_name
	LogicalJoin Single
		Project* person.first_name
			Filter* (Equal person.person_id customer.person_id)
				LogicalGet person
		LogicalJoin Single
			Project* store.name
				Filter* (Equal store.store_id customer.customer_id) (Like store.name A%)
					LogicalGet store
			Project* customer.customer_id customer.person_id
				LogicalGet customer
