select (select name from store where store.store_id = customer.customer_id and store.name like "A%"), (select first_name from person where person.person_id = customer.person_id) 
from customer

LogicalMap $query.$col1:store.name $query.$col2:person.first_name
	LogicalJoin Single (NaturalJoin customer.person_id)
		LogicalMap person.first_name customer.person_id
			LogicalJoin Inner (Equal person.person_id customer.person_id)
				Map* person.person_id person.first_name
					LogicalGet person
				LogicalProject customer.person_id
					LogicalJoin Single (NaturalJoin customer.customer_id)
						LogicalMap store.name customer.customer_id
							LogicalJoin Inner (Equal store.store_id customer.customer_id)
								Filter* (Like store.name A%)
									Map* store.store_id store.name
										LogicalGet store
								LogicalProject customer.customer_id
									Map* customer.customer_id customer.person_id
										LogicalGet customer
						Map* customer.customer_id customer.person_id
							LogicalGet customer
		LogicalJoin Single (NaturalJoin customer.customer_id)
			LogicalMap store.name customer.customer_id
				LogicalJoin Inner (Equal store.store_id customer.customer_id)
					Filter* (Like store.name A%)
						Map* store.store_id store.name
							LogicalGet store
					LogicalProject customer.customer_id
						Map* customer.customer_id customer.person_id
							LogicalGet customer
			Map* customer.customer_id customer.person_id
				LogicalGet customer
