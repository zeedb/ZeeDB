select person_id
from person
where modified_date = (select max(modified_date) from person)

Map person.person_id
	Filter (Equal person.modified_date $aggregate.$agg1)
		NestedLoop Single
			Aggregate $aggregate.$agg1:(Max $proj.$max)
				Map $proj.$max:person.modified_date
					Filter* (LessOrEqual person.$xmin (Xid)) (Less (Xid) person.$xmax)
						Map* person.modified_date person.$xmin person.$xmax
							SeqScan person
			Filter* (LessOrEqual person.$xmin (Xid)) (Less (Xid) person.$xmax)
				Map* person.person_id person.modified_date person.$xmin person.$xmax
					SeqScan person
