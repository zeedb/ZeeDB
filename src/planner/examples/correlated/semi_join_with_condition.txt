select 1
from person, store
where person.person_id in (select person_id from customer where customer.store_id = store.store_id)

Map $query.$col1:1
	HashJoin Semi customer.person_id=person.person_id customer.store_id=store.store_id
		Map* customer.person_id customer.store_id
			SeqScan customer
		NestedLoop Inner
			Map* store.store_id
				SeqScan store
			Map* person.person_id
				SeqScan person
