set first_name = (select last_name)
where person_id = 1

	Map person.person_id person.first_name:$expr_subquery.last_name person.last_name person.modified_date person.$pid person.$tid
		NestedLoop Single (Equal person.last_name person.last_name')
			Map $expr_subquery.last_name:person.last_name' person.last_name'
				NestedLoop Inner
					Aggregate person.last_name'
						Filter* (LessOrEqual person.$xmin (Xid)) (Less (Xid) person.$xmax) (Equal person.person_id 1)
							Map* person.person_id person.first_name person.last_name' person.modified_date person.$xmin person.$xmax person.$pid person.$tid
								SeqScan person
					TableFreeScan
			Filter* (LessOrEqual person.$xmin (Xid)) (Less (Xid) person.$xmax) (Equal person.person_id 1)
				Map* person.person_id person.first_name person.last_name person.modified_date person.$xmin person.$xmax person.$pid person.$tid
					SeqScan person
