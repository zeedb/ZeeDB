update customer
set account_number = 0
where person_id in (select person_id from person where first_name = "Joe")

Update customer.account_number:$update.$expr
	Map $update.$expr:0
		HashJoin Semi (Equal customer.person_id person.person_id)
			Partition* person.person_id
				Filter* (LessOrEqual person.$xmin (Xid)) (Less (Xid) person.$xmax) (Equal person.first_name "Joe")
					Map* person.person_id person.first_name person.$xmin person.$xmax
						SeqScan person
			Partition* customer.person_id
				Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
					Map* customer.person_id customer.account_number customer.$xmin customer.$xmax
						SeqScan customer
