select first_name
from person
where person_id in (select person_id from customer) order by modified_date limit 10

Limit 10 0
	Sort person.modified_date
		Map $mark.$in:true customer.person_id person.first_name person.modified_date person.person_id
			HashJoin Semi (Equal person.person_id customer.person_id)
				Partition* customer.person_id
					Map* customer.person_id
						SeqScan customer
				Partition* person.person_id
					Map* person.person_id person.first_name person.modified_date
						SeqScan person
