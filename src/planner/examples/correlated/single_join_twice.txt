select (select name from store where store.store_id = customer.customer_id), (select first_name from person where person.person_id = customer.person_id)
from customer

Map $query.$col1:store.name $query.$col2:person.first_name
	HashJoin Single (Equal customer.person_id customer.person_id')
		Partition* customer.person_id'
			Map person.first_name customer.person_id'
				Map customer.person_id':person.person_id, ..
					Map* person.person_id person.first_name
						SeqScan person
		Partition* customer.person_id
			HashJoin Single (Equal customer.customer_id customer.customer_id')
				Partition* customer.customer_id'
					Map store.name customer.customer_id'
						Map customer.customer_id':store.store_id, ..
							Map* store.store_id store.name
								SeqScan store
				Partition* customer.customer_id
					Map* customer.customer_id customer.person_id
						SeqScan customer
