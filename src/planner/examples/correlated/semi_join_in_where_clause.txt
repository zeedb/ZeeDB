select customer_id
from customer c1, store where c1.store_id = store.store_id
and c1.modified_date in (select modified_date from customer c2 where c2.account_number > c1.account_number)

Map customer.customer_id
	HashJoin Semi (Equal customer.account_number customer.account_number') (Equal customer.modified_date customer.modified_date)
		Partition* customer.account_number' customer.modified_date
			Map customer.modified_date customer.account_number'
				NestedLoop Inner (Greater customer.account_number customer.account_number')
					Aggregate customer.account_number'
						NestedLoop Inner
							Map* store.store_id
								SeqScan store
							Map* customer.customer_id customer.store_id customer.account_number' customer.modified_date
								SeqScan customer
					Map* customer.account_number customer.modified_date
						SeqScan customer
		Partition* customer.account_number customer.modified_date
			HashJoin Inner (Equal customer.store_id store.store_id)
				Partition* store.store_id
					Map* store.store_id
						SeqScan store
				Partition* customer.store_id
					Map* customer.customer_id customer.store_id customer.account_number customer.modified_date
						SeqScan customer
