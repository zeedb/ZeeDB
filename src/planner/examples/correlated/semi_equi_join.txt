select 1
from person
where person_id in (select person_id from customer where person.modified_date = customer.modified_date)

Map $query.$col1:1
	HashJoin Semi (Equal person.modified_date person.modified_date') (Equal person.person_id customer.person_id)
		Partition* person.modified_date' customer.person_id
			Map customer.person_id person.modified_date'
				Map person.modified_date':customer.modified_date, ..
					Filter* (LessOrEqual customer.$xmin (Xid)) (Less (Xid) customer.$xmax)
						Map* customer.person_id customer.modified_date customer.$xmin customer.$xmax
							SeqScan customer
		Partition* person.modified_date person.person_id
			Filter* (LessOrEqual person.$xmin (Xid)) (Less (Xid) person.$xmax)
				Map* person.person_id person.modified_date person.$xmin person.$xmax
					SeqScan person
